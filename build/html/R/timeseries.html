# 计算累计三、五、十、二十、三十天的移动净流入和

ruyi2['summa10'] <- SMA(ruyi2$net_mf_amount, 10) * 10
ruyi2['summa40'] <- SMA(ruyi2$net_mf_amount, 40) * 40
